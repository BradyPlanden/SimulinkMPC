31void mpc(mwSignedIndex T, mwSignedIndex m, mwSignedIndex n, mwSignedIndex niters, double kappa,

41    



42    



43    nz = T*(n+m);



44    



45    



46    /* Dynamic Allocation -- Consider changing to Static for runtime performance*/



47    At = malloc(sizeof(double)*n*n);



48    Bt = malloc(sizeof(double)*n*m);



49    eyen = malloc(sizeof(double)*n*n);



50    eyem = malloc(sizeof(double)*m*m);

52    x = malloc(sizeof(double)*n);



53    zmax = malloc(sizeof(double)*nz);



54    zmin = malloc(sizeof(double)*nz);



55    zmaxp = malloc(sizeof(double)*nz);



56    zminp = malloc(sizeof(double)*nz);



57    



58    /* eyen, eyem */



59    dptr = eyen;



60    for (i = 0; i < n*n; i++)



61    {



62        *dptr = 0;



63        dptr++;



64    }



65    dptr = dptr-n*n;



66    for (i = 0; i < n; i++)



67    {



68        *dptr = 1;

70    }

72    dptr = eyem;



73    for (i = 0; i < m*m; i++)



74    {



75        *dptr = 0;



76        dptr++;



77    }

79    for (i = 0; i < m; i++)



80    {



81        *(dptr+i*m+i) = 1;



82    }



83    dptr = x; dptr1 = x0;



84    for (i = 0; i < n; i++)



85    {



86        *dptr = *dptr1;



87        dptr++; dptr1++;



88    }



89    dptr = z;



90    for (i = 0; i < T; i++)



91    {



92        for (j = 0; j < m; j++)



93        {



94            *dptr = *(U0+i*m+j);



95            dptr++;



96        }



97        for (j = 0; j < n; j++)



98        {



99            *dptr = *(X0+i*n+j);



100            dptr++;



101        }



102    }



103    



104    /* At, Bt */



105    dgemm("t","n",&n,&n,&n,&fone,A,&n,eyen,&n,&fzero,At,&n);



106    dgemm("n","t",&m,&n,&m,&fone,eyem,&m,B,&n,&fzero,Bt,&m);

109



110    /* zmax, zmin */



111    dptr1 = zmax;



112    dptr2 = zmin;



113    for (i = 0; i < T; i++)



114    {



115        for (j = 0; j < m; j++)



116        {



117            *dptr1 = *(umax+j);



118            *dptr2 = *(umin+j);



119            dptr1++; dptr2++;



120        }

132    for (i = 0; i < nz; i++) zmaxp[i] = zmax[i] - 0.01*(zmax[i]-zmin[i]);



133



134    /* project z */



135    for (i = 0; i < nz; i++) z[i] = z[i] > zmaxp[i] ? zmaxp[i] : z[i]; // Loop through Z and constrain values to defined maximum



136    for (i = 0; i < nz; i++) z[i] = z[i] < zminp[i] ? zminp[i] : z[i]; // Loop through Z and constrain values to defined minimum



137    



138    



139    /* Call fmpc*/



140    fmpcsolve(A,B,At,Bt,eyen,eyem,Q,R,Qf,zmax,zmin,x,z,T,n,m,nz,niters,kappa);



141    



142    dptr = z;



143    for (i = 0; i < T; i++)



144    {



145        for (j = 0; j < m; j++)



146        {



147            *(U+i*m+j) = *dptr;



148            dptr++;



149        }



150        for (j = 0; j < n; j++)



151        {



152            *(X+i*n+j) = *dptr;



153            dptr++;



154        }



155    }



156



157    free(At); free(Bt); free(eyen); free(eyem);



158    free(z); free(x); free(zmax); free(zmin);



159    return;  



160

166void fmpcsolve(double *A, double *B, double *At, double *Bt, double *eyen,



167         double *eyem, double *Q, double *R, double *Qf, double *zmax, double *zmin, 



168         double *x, double *z0, mwSignedIndex T, mwSignedIndex n, mwSignedIndex m, mwSignedIndex nz, mwSignedIndex niters, double kappa) 



169{



170    mwSignedIndex maxiter = niters;



171    int iiter, i, cont;



172    double alpha = 0.01;



173    double beta = 0.9;



174    double tol = 0.1;



175    double s;



176    double resd, resp, res, newresd, newresp, newres;



177    double *b, *z, *nu, *Ctnu;



178    double *dnu, *dz;



179    double *gf, *gp, *hp, *newgf, *newgp, *newhp;



180    double *rd, *rp, *newrd, *newrp;



181    double *dptr, *dptr1, *dptr2, *dptr3;



182    double *newnu, *newz, *newCtnu;



183



184    /* memory allocation */



185    b = malloc(sizeof(double)*T*n);



186    dnu = malloc(sizeof(double)*T*n);



187    dz = malloc(sizeof(double)*nz);



188    nu = malloc(sizeof(double)*T*n);



189    Ctnu = malloc(sizeof(double)*nz);



190    z = malloc(sizeof(double)*nz);



191    gp = malloc(sizeof(double)*nz);



192    hp = malloc(sizeof(double)*nz);



193    gf = malloc(sizeof(double)*nz);



194    rp = malloc(sizeof(double)*T*n);



195    rd = malloc(sizeof(double)*nz);



196    newnu = malloc(sizeof(double)*T*n);



197    newz = malloc(sizeof(double)*nz);



198    newCtnu = malloc(sizeof(double)*nz);



199    newgf = malloc(sizeof(double)*nz);



200    newhp = malloc(sizeof(double)*nz);



201    newgp = malloc(sizeof(double)*nz);



202    newrp = malloc(sizeof(double)*T*n);



203    newrd = malloc(sizeof(double)*nz);



204



205    for (i = 0; i < n*T; i++) nu[i] = 0;



206    for (i = 0; i < n*T; i++) b[i] = 0;



207    dgemv("n",&n,&n,&fone,A,&n,x,&ione,&fzero,b,&ione);



208   // dgemv(CblasRowMajor,CblasNoTrans,n,n,fone,A,n,x,ione,fzero,b,ione); // Computes a real, double precision matrix-vector product using a general matrix.



209    dptr = z; dptr1 = z0;



210    for (i = 0; i < nz; i++) 



211    {



212        *dptr = *dptr1;



213        dptr++; dptr1++;



214    }



215    if (quiet == 0)



216    {   



217        printf("\n iteration \t step \t\t rd \t\t\t rp\n");



218    }



219    for (iiter = 0; iiter < maxiter; iiter++)



220    {



221        gfgphp(Q,R,Qf,zmax,zmin,z,T,n,m,nz,gf,gp,hp);



222        rdrp(A,B,Q,R,Qf,z,nu,gf,gp,b,T,n,m,nz,kappa,rd,rp,Ctnu);



223        resdresp(rd,rp,T,n,nz,&resd,&resp,&res);



224        if (res < tol) break;



225



226        dnudz(A,B,At,Bt,eyen,eyem,Q,R,Qf,hp,rd,rp,T,n,m,nz,kappa,dnu,dz); 



227



228        s = 1;



229        



230        /* feasibility search */



231        while (1)



232        {



233            cont = 0;



234            dptr = z; dptr1 = dz; dptr2 = zmax; dptr3 = zmin;



235            for (i = 0; i < nz; i++)



236            {



237                if (*dptr+s*(*dptr1) >= *dptr2) cont = 1;



238                if (*dptr+s*(*dptr1) <= *dptr3) cont = 1;



239                dptr++; dptr1++; dptr2++; dptr3++;



240            }



241            if (cont == 1)



242            {



243                s = beta*s;



244                continue;



245            }



246            else



247                break;



248        }



249        dptr = newnu; dptr1 = nu; dptr2 = dnu;



250        for (i = 0; i < T*n; i++)



251        {



252            *dptr = *dptr1+s*(*dptr2);



253            dptr++; dptr1++; dptr2++;



254        }



255        dptr = newz; dptr1 = z; dptr2 = dz;



256        for (i = 0; i < nz; i++)



257        {



258            *dptr = *dptr1+s*(*dptr2);



259            dptr++; dptr1++; dptr2++;



260        }



261



262        /* insert backtracking line search */



263        while (1)



264        {



265            gfgphp(Q,R,Qf,zmax,zmin,newz,T,n,m,nz,newgf,newgp,newhp);



266            rdrp(A,B,Q,R,Qf,newz,newnu,newgf,newgp,b,T,n,m,nz,kappa,newrd,newrp,newCtnu);



267            resdresp(newrd,newrp,T,n,nz,&newresd,&newresp,&newres);



268            if (newres <= (1-alpha*s)*res) break;



269            s = beta*s;



270            dptr = newnu; dptr1 = nu; dptr2 = dnu;



271            for (i = 0; i < T*n; i++)



272            {



273                *dptr = *dptr1+s*(*dptr2);



274                dptr++; dptr1++; dptr2++;



275            }



276            dptr = newz; dptr1 = z; dptr2 = dz;



277            for (i = 0; i < nz; i++)



278            {



279                *dptr = *dptr1+s*(*dptr2);



280                dptr++; dptr1++; dptr2++;



281            }



282        }



283        



284        dptr = nu; dptr1 = newnu; 



285        for (i = 0; i < T*n; i++)



286        {



287            *dptr = *dptr1;



288            dptr++; dptr1++;



289        }



290        dptr = z; dptr1 = newz;



291        for (i = 0; i < nz; i++)



292        {



293            *dptr = *dptr1;



294            dptr++; dptr1++;



295        }



296        if (quiet == 0)



297        {



298            printf("    %d \t\t %5.4f \t %0.5e \t\t %0.5e\n",iiter,s,newresd,newresp);



299        }



300    }



301    dptr = z0; dptr1 = z;



302    for (i = 0; i < nz; i++)



303    {



304        *dptr = *dptr1;



305        dptr++; dptr1++;



306    }



307



308    free(b); free(dnu); free(dz); free(nu); free(Ctnu);



309    free(z); free(gp); free(hp); free(gf); free(rp); free(rd);



310    free(newnu); free(newz); free(newCtnu); free(newgf); free(newhp);



311    free(newgp); free(newrp); free(newrd);



312    return;



313}

316void dnudz(double *A, double *B, double *At, double *Bt, double *eyen,

378    for (j = 0; j < m*m; j++)



379    {



380        *dptr = 2*(*dptr1);



381        dptr++; dptr1++;



382    }

384    for (j = 0; j < m; j++)



385    {



386        *dptr = *dptr+kappa*(*dptr1);

388    }

394    }

469        for (j = 0; j < n*n; j++)



470        {



471            *dptr = *dptr1;



472            dptr++; dptr1++;



473        }

477        dptr1 = PhiinvQAt+n*n*(i-1);

479        //dgemm(CblasRowMajor,CblasNoTrans,CblasNoTrans,n,n,n,fone,A,n,dptr1,n,fone,dptr,n); // Computes a real, double precision, matrix-matrix product with general matrices.



480    }



481

483    dptr = Ld; dptr1 = Yd; 



484    for (i = 0; i < n*n; i++)
